% Custom command to create a form field with vertical spacing and manual width
\newcommand{\formfield}[3]{%
  \fieldspace
  \noindent\textbf{#1:}%
  \begin{tikzpicture}[baseline=(field.base)]
    \node[anchor=base west, outer sep=0pt, inner sep=0pt] (field) at (0,0) {\TextField[name=#2, width=\dimexpr\basefieldlength-#3\relax, bordercolor={1 1 1}, charsize=12pt]{}};
    \draw[black] (field.base west) ++(0,-2pt) -- ++(\dimexpr\basefieldlength-#3\relax,0); % Underline slightly below the field
  \end{tikzpicture}%
}

% Custom command to create an address field with vertical spacing and manual width
\newcommand{\addressfield}[2]{%
  \addressspace
  \noindent
  \begin{tikzpicture}[baseline=(field.base)]
    \node[anchor=base west, outer sep=0pt, inner sep=0pt] (field) at (0,0) {\TextField[name=#1, width=\dimexpr\basefieldlength-#2\relax, bordercolor={1 1 1}, charsize=12pt]{}};
    \draw[black] (field.base west) ++(0,-2pt) -- ++(\dimexpr\basefieldlength-#2\relax,0); % Underline slightly below the field
  \end{tikzpicture}%
}
